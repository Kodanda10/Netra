export function genSeries(n = 24, base = 100) {
  const arr: { t: number; v: number }[] = []
  let v = base
  for (let i = 0; i < n; i++) { v += (Math.random() - 0.5) * 8; arr.push({ t: i, v: Number(v.toFixed(2)) }) }
  return arr
}
